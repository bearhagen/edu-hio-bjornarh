function oddCount(arr) {
	var res = 0;
	arr.forEach(function(i, val) {
		if (val%3 === 0) {
			res += val;
		}
	});

	return "Oddetall: " + res;
}

function evenCount(arr) {
	var res = 0;
	arr.forEach(function(i, val) {
		if (val%2 === 0) {
			res += val;
		}
	});

	return "Oddetall: " + res;	
}

var tall = [3, 42, 7, 69, 12, 9, 37, 17, 10];
document.getElementById("utskrift").innerHTML = oddCount(tall) + "<br>" + evenCount(tall);
